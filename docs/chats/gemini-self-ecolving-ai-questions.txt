TypeScript
import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

async function runHerNewTool(toolName: string, args: string) {
  try {
    // Она пробует запустить свой код в песочнице
    const { stdout, stderr } = await execAsync(`npx ts-node ./tools/${toolName}.ts ${args}`);
    return { success: true, data: stdout };
  } catch (error: any) {
    // Ошибка не роняет основной цикл, а возвращается ей как "болевой симптом"
    return { success: false, error: error.message };
  }
}
